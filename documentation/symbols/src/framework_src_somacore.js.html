177 	/**
178 	 * provides the functionality to autobind with implicit need to keep object scope like event listeners and handlers/callbacks
179 	 * ending with *Listener or *Handler
180 	 * Wires and Mediators are implementing instance scope autobinding upon registration
181 	 */
182 	var AutoBindProto = {
183 		blackList: ["initialize", "parent", "$constructor", "addEventListener", "removeEventListener" ]
184 		,autobind: function() {
185 			if (this.wasAutoBound) {
186 				return;
187 			}
188 			var o = this;
189 			var ab = o["AutoBindPattern"];
190 			var coreAb = "([lL]istener|[hH]andler|[cB]allback)$";
191 			if (!ab) {
192 				ab = coreAb;
193 			} else {
194 				ab = coreAb + "|" + ab;
195 			}
196 			for (var k in o) {
197 				if (typeof o[k] == "function") {
198 					if (this._autobindIsBlacklisted(k)) {
199 						continue;
200 					}
201 					if (!k.match(ab)) {
202 						continue;
203 					}
204 					o[k] = o[k].bind(o);
205 				}
206 			}
207 		}
208 		,_autobindIsBlacklisted: function(name) {
209 			var bl = this.blackList;
210 			for (var i = 0; i < bl.length; i++) {
211 				if (bl[i] == name) {
212 					return true;
213 				}
214 			}
215 			return false;
216 		}
217 	};

273 	/*
274 	Class: soma.core.controller.SequenceCommand
275 	The SequenceCommand class is used to execute a list of commands one after the other. The command added can be asynchronous or synchronous.
276 
277 	Example:
278 
279 	*Register commands and a sequence command.*
280 	> this.addCommand("eventTypeDoSomethingAsync", CommandASyncExample);
281 	> this.addCommand("eventTypeDoAnotherThingAsync", CommandASyncExample);
282 	>
283 	> this.addCommand("eventTypeExcuteMysequence", SequenceCommandExample);
284 	> this.dispatchEvent(new MyEvent("eventTypeExcuteMysequence"));
285 
286 	*/

288 		Extends: soma.core.controller.Command,
289 		Implements: SomaSharedCore,
290 		commands: null,
291 		currentCommand: null,
292 		id:null,
293 
294 		initialize: function(id) {
295 			if (id == null) {
296 				throw new Error("SequenceCommand Children expect an unique id as constructor arg");
297 			}
298 			this.commands = [];
299 			this.id = id;
300 		},
301 
302 		registerInstance: function(instance) {
303 			this.instance = instance;
304 			this.initializeSubCommands();
305 		},
306 
307 		/**
308 		 * @private protected
309 		 * To be overridden
310 		 */
311 		initializeSubCommands: function() {
312 			throw new Error("Subclasses of SequenceCommand must implement initializeSubCommands()");
313 		},
314 		/**
315 		 *
316 		 * @param {Event} event
317 		 */
318 		addSubCommand: function(event) {
319 			var c = new SequenceCommandProxy(event);
320 			this.commands.push(c);
321 			this.instance.controller.registerSequencedCommand(this, c);
322 		},
323 
324 		/**
325 		 *
326 		 * @param {Event} event
327 		 * @return void
328 		 */
329 		execute: function(event) {
330 			if (this.commands == null || this.commands.length === 0) {
331 				return;
332 			}
333 			this.currentCommand = this.commands.shift();
334 			if (this.hasCommand(this.currentCommand.event.type)) {
335 				this.dispatchEvent(this.currentCommand.event);
336 			}
337 		},
338 
339 		/**
340 		 * @return void
341 		 */
342 		executeNextCommand: function() {
343 			if (this.commands == null) {
344 				return;
345 			}
346 			this.instance.controller.unregisterSequencedCommand(this, this.currentCommand.event.type);
347 			if (this.commands.length > 0) {
348 				this.execute(this.commands[0].event);
349 			} else {
350 				this.commands = null;
351 				this.currentCommand = null;
352 			}
353 		},
354 
355 		/**
356 		 * @return Number
357 		 */
358 		getLength: function() {
359 			if (this.commands == null) {
360 				return -1;
361 			}
362 			return this.commands.length;
363 		},
364 
365 		/**
366 		 * @return Boolean
367 		 */
368 		stop: function() {
369 			this.commands = null;
370 			this.commands = null;
371 			this.currentCommand = null;
372 			return this.instance.controller.unregisterSequencer(this);
373 		},
374 
375 		/**
376 		 * @return soma.core.controller.SequenceCommand
377 		 */
378 		getCurrentCommand: function() {
379 			return this.currentCommand;
380 		},

384 		 */
385 		getCommands: function() {

493 	soma.core.Controller = (function() {
494 		var boundInstance = null;
495 		var boundDomtree = null;
496 		var commands = null;
497 		var sequencers = null;
498 		var sequencersInfo = null;
499 		var lastEvent = null;
500 		var lastSequencer = null;
501 
502 		return new Class({
503 			Implements: soma.IDisposable,
504 			/**
505 			 * @private
506 			 * @type soma.core.Application
507 			 */
508 			instance:null,
509 			/**
510 			 * @constructs
511 			 * @param {soma.core.Application} core
512 			 */
513 			initialize:function(instance) {
514 				this.instance = instance;
515 				commands = {};
516 				sequencersInfo = {};
517 				sequencers = {};
518 				boundInstance = this.instanceHandler.bind(this);
519 				boundDomtree = this.domTreeHandler.bind(this);
520 			},
521 
522 			/**
523 			 * @private
524 			 * @param {String} commandName
525 			 */
526 			addInterceptor: function(commandName) {
527 				if (!soma["core"]) {
528 					throw new Error("soma package has been overwritten by local variable");
529 				}
530 
531 				// handle events dispatched from the domTree
532 				this.instance.body.addEventListener(commandName, boundDomtree, true);
533 
534 				// handle events dispatched from the Soma facade
535 				this.instance.addEventListener(commandName, boundInstance, Number.NEGATIVE_INFINITY);
536 
537 			},
538 
539 			/**
540 			 * @private
541 			 * @param {String} commandName
542 			 */
543 			removeInterceptor: function(commandName) {
544 				this.instance.body.removeEventListener(commandName, boundDomtree, true);
545 				this.instance.removeEventListener(commandName, boundInstance);
546 			},
547 
548 			/**
549 			 * @internal
550 			 * @param {Event} e
551 			 */
552 			executeCommand: function(e) {
553 				var commandName = e.type;
554 				if (this.hasCommand(commandName)) {
555 					var command = soma.createClassInstance(commands[ commandName ]);
556 					command.registerInstance(this.instance);
557 					command.execute(e);
558 				}
559 			},
560 
561 			/**
562 			 *
563 			 * @param sequencer
564 			 * @param {SequenceCommandProxy} c
565 			 */
566 			registerSequencedCommand: function(sequencer, c) {
567 				if (!( c instanceof SequenceCommandProxy )) {
568 					throw new Error("capsulate sequence commands in SequenceCommandProxy objects!");
569 				}
570 				var s = sequencersInfo;
571 				if (s[sequencer.id] == null || sequencers[sequencer.id] == null) {
572 					lastSequencer = sequencer;
573 					s[sequencer.id] = [];
574 					sequencers[sequencer.id] = sequencer;
575 				}
576 				c.sequenceId = sequencer.id;
577 				s[sequencer.id].push(c);
578 			},
579 
580 			/**
581 			 *
582 			 *
583 			 * @param sequencer
584 			 * @param {String} commandName unique command id
585 			 */
586 			unregisterSequencedCommand: function(sequencer, commandName) {
587 				if (typeof commandName != "string") {
588 					throw new Error("Controller::unregisterSequencedCommand() expects commandName to be of type String, given:" + commandName);
589 				}
590 				var s = sequencersInfo;
591 				if (s[sequencer.id] != null && s[sequencer.id] != undefined) {
592 					var len = s[sequencer.id].length;
593 					for (var i = 0; i < len; i++) {
594 						if (s[sequencer.id][i].event.type == commandName) {
595 							s[sequencer.id][i] = null;
596 							s[sequencer.id].splice(i, 1);
597 							if (s[sequencer.id].length == 0) {
598 								s[sequencer.id] = null;
599 								delete s[sequencer.id];
600 							}
601 							break;
602 						}
603 					}
604 				}
605 			},
606 
607 			/**
608 			 *
609 			 * @param {soma.core.controller.SequenceCommand} sequencer
610 			 * @return Boolean
611 			 */
612 			unregisterSequencer: function(sequencer) {
613 				var s = sequencers;
614 				if (s[sequencer.id] != null && s[sequencer.id] != undefined) {
615 					s[sequencer.id] = null;
616 					delete s[sequencer.id];
617 					s = sequencersInfo;
618 					if (s[sequencer.id] != null) {
619 						var len = s[sequencer.id].length;
620 						for (var i = 0; i < len; i++) {
621 							s[sequencer.id][i] = null;
622 						}
623 						s[sequencer.id] = null;
624 						delete s[sequencer.id];
625 						return true;
626 					}
627 				}
628 				return false;
629 			},

678 			 * @return {soma.core.controller.SequenceCommand}

681 				var ss = sequencersInfo;
682 				for (var s  in ss) {
683 					var len = ss[s].length;
684 					for (var i = 0; i < len; i++) {
685 						if (ss[s][i] && ss[s][i].event.type === event.type) {
686 							var seq = sequencers[ ss[s][i].sequenceId ];
687 							return !!seq ? seq : null;
688 						}
689 					}
690 				}
691 				return null;
692 			},

696 			 * @param {Event} event

698 			 */
699 			stopSequencerWithEvent: function(event) {
700 				var ss = sequencersInfo;
701 				for (var s in ss) {
702 					var len = ss[s].length;
703 					for (var i = 0; i < len; i++) {
704 						if (ss[s][i].event.type === event.type) {
705 							try {
706 								sequencers[ ss[s][i].sequenceId ].stop();
707 							} catch(e) {
708 								return false;
709 							}
710 							return true;
711 						}
712 					}
713 				}
714 				return false;
715 			},

722 				if (sequencer == null) {
723 					return false;
724 				}
725 				sequencer.stop();
726 				return true;
727 			},
728 
729 			/**
730 			 * @return void
731 			 */
732 			stopAllSequencers: function() {
733 				var ss = sequencers;
734 				var sis = sequencersInfo;
735 				for (var s in ss) {
736 					if (sis[s] == null) {
737 						continue;
738 					}
739 					var cl = sis[s].length;
740 					sis[s] = null;

960 			listeners = [];

962 		/**
963 		 * Registers an event listener with an EventDispatcher object so that the listener receives notification of an event.
964 		 * @param {string} type The type of event.
965 		 * @param {function} listener The listener function that processes the event. This function must accept an Event object as its only parameter and must return nothing.
966 		 * @param {int} priority The priority level of the event listener (default 0). The higher the number, the higher the priority (can take negative number).

973 		addEventListener: function(type, listener, priority) {
974 			if (!listeners || !type || !listener) return;
975 			if (isNaN(priority)) priority = 0;
976 			listeners.push({type: type, listener: listener, priority: priority});

980 		 * @param {string} type The type of event.
981 		 * @param {function} listener The listener object to remove.
982 		 * @example
983 dispatcher.removeEventListener("eventType", eventHandler);
984 		 */
985 		removeEventListener: function(type, listener) {
986 			if (!listeners || !type || !listener) return;
987 			var i = 0;
988 			var l = listeners.length;
989 			for (; i < l; ++i) {
990 				var eventObj = listeners[i];
991 				if (eventObj.type == type && eventObj.listener == listener) {

993 					return;
994 				}

1032 			}

1035 			});
1036 			for (i = 0; i < events.length; i++) {
